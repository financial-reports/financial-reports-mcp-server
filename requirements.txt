# Direct dependencies only — pip resolves the rest.

# MCP framework — FastMCP. The auth module is unstable across minor versions per
# upstream policy, and the OAuth proxy's DCR client store is the load-bearing part:
# a minor bump that changes the `mcp-oauth-proxy-clients` collection name, the key
# format, or the ProxyDCRClient model makes every existing registration deserialize
# to None. That reaches users as "client not found" and forces every connector to
# be removed and re-added. A range re-resolved on every image rebuild, so dev ran
# 2.13.3 while prod ran 2.14.7. Keep this exact; bump deliberately, and diff
# fastmcp/server/auth/oauth_proxy.py first. 3.x exists — do not move without an audit.
fastmcp==2.14.7

# HTTP host
fastapi>=0.115.0
uvicorn>=0.32.0

# Backend HTTP client. The [http2] extra pulls in `h2` so the module-level
# AsyncClient can multiplex parallel tool calls (Claude routinely fans out
# 5–10 simultaneous calls per session) over a single TCP/TLS connection
# to api.financialreports.eu.
httpx[http2]>=0.28.0

# Subscription verification cache
cachetools>=5.3.0

# Redis client — used by FastMCP's OAuth state storage when MCP_REDIS_URL is
# set. Without it, FastMCP falls back to a per-replica encrypted DiskStore,
# which is wiped on container restart and not shared across replicas. With it,
# refresh tokens persist across deploys and across horizontally scaled
# replicas. `py-key-value-aio` is already a transitive dep via FastMCP; only
# the redis client itself needs to be declared.
redis>=5.0,<8

# Schema generator (build-time only, but kept here so the Docker builder has them)
PyYAML>=6.0
Jinja2>=3.1

# SDK to FinancialReports backend.
# This pin is auto-bumped by the `Generate and Publish SDKs` workflow in the
# financial-reports/web repo (.github/workflows/generate_sdks.yml on master)
# whenever the OpenAPI schema changes. That workflow also rebuilds and
# redeploys this MCP server's image to Azure Container Apps.
financial_reports_generated_client==1.4.49
